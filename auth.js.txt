// auth.js

// Helpers to read/write auth
function getAuthToken() {
  return localStorage.getItem('nox_auth_token');
}

function getAuthUser() {
  const raw = localStorage.getItem('nox_user');
  if (!raw) return null;
  try {
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

function clearAuth() {
  localStorage.removeItem('nox_auth_token');
  localStorage.removeItem('nox_user');
}

// Render navbar auth section
function renderNavAuth() {
  const navAuth = document.getElementById('navAuth');
  if (!navAuth) return;

  const token = getAuthToken();

  if (token) {
    // Logged in: show Profile + Logout
    navAuth.innerHTML = `
      <a href="profile.html"
         class="text-sm font-medium text-gray-600 hover:text-[#0B1F3A]">
        Profile
      </a>
      <button
        id="logoutBtn"
        class="px-4 py-2 text-sm font-medium text-white rounded-full gradient-btn shadow-sm">
        Logout
      </button>
    `;
  } else {
    // Logged out: show Sign up button
    navAuth.innerHTML = `
      <a href="signup.html"
         class="gradient-btn px-6 py-2.5 text-sm font-medium text-white rounded-full shadow-sm inline-flex items-center gap-2">
        Sign up
        <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2"
                d="M17 8l4 4m0 0l-4 4m4-4H3" />
        </svg>
      </a>
    `;
  }

  attachLogoutHandler();
}

// Attach logout modal behavior
function attachLogoutHandler() {
  const logoutBtn = document.getElementById('logoutBtn');
  const logoutModal = document.getElementById('logoutModal');
  if (!logoutBtn || !logoutModal) return;

  const cancelBtn = document.getElementById('cancelLogoutBtn');
  const confirmBtn = document.getElementById('confirmLogoutBtn');

  logoutBtn.addEventListener('click', () => {
    logoutModal.classList.remove('hidden');
    logoutModal.classList.add('flex');
  });

  cancelBtn.addEventListener('click', () => {
    logoutModal.classList.add('hidden');
    logoutModal.classList.remove('flex');
  });

  confirmBtn.addEventListener('click', () => {
    clearAuth();
    logoutModal.classList.add('hidden');
    logoutModal.classList.remove('flex');
    window.location.href = 'index.html';
  });
}

// Protect profile page: if not logged in, redirect to signup
function protectProfilePage() {
  if (window.location.pathname.endsWith('profile.html')) {
    const token = getAuthToken();
    if (!token) {
      window.location.href = 'signup.html';
    }
  }
}

// Populate profile info if on profile page
function populateProfileInfo() {
  if (!window.location.pathname.endsWith('profile.html')) return;
  const user = getAuthUser();
  if (!user) return;

  const nameEl = document.getElementById('profileName');
  const emailEl = document.getElementById('profileEmail');

  if (nameEl) nameEl.textContent = user.name || 'Nox user';
  if (emailEl) emailEl.textContent = user.email || 'email not set';
}

// Init on DOM ready
document.addEventListener('DOMContentLoaded', () => {
  renderNavAuth();
  protectProfilePage();
  populateProfileInfo();
});
